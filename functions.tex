\subsection*{Functions}  
\begin{frame}{What is a function?}
    \begin{itemize}
        \item Series of evaluations with defined inputs adn outputs
        \item Allows for code to be organized
        \item Deterministic - fixed input gives a fixed output
        \item Easier to test
        \item Easier to manipulate and use - efficient building block for more complex programs
    \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Built-in Functions}
    \begin{itemize}
        \item Before using a funciton you must either define it or import it into Python
        \item Read the documentation of a function to understand what to use it - inside Ipython
            \begin{verbatim}
            import numpy
            numpy.sin?
        \end{verbatim}
    \item Calling a function is easy - \texttt{out = function(in)}
    \item Many already built into Python - others inside packages (Numpy, Scipy, your own, etc.)
    \end{itemize}

\end{frame}

\begin{frame}\frametitle{How do you write a function?}
    \begin{itemize}
        \item Functions can be written in any text editor
        \item Only need to use spaces (instead of Tabs)
        \item Best to have a ``good'' editor
            \begin{itemize}
                \item Automatic indentation - very important in Python
                \item Syntax highlighting - pretty colors
                \item Autocompletion - for the lazy
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Example Function}
\begin{verbatim}

    """Documentation

    output_1 = input_1

    return output_1, output_2

output_1, output_2 = function_name(input_1, input_2)

\end{frame}

\begin{frame}{Function scope}
    \begin{itemize}
        \item Variable available inside of a function
            \begin{itemize}
                \item Inputs passed into functions
                \item Module level variables
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Script vs. Function}
    \begin{itemize}
        \item Function requires inputs - must be called properly
        \item Scripts - can be run interactively
        \item Variables available to interactive interpreter
    \end{itemize}
\end{frame}
